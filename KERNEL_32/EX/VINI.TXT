проц init()
  цвет:=0
  растяжение:=.л.
  видимость:=.и.
  слева:=0
  сверху:=0
  скорость:=10
кон проц

проц телепортироваться_в(х,у)
  слева:=ширина*х
  сверху:=высота*у
кон проц

проц спросить_мёд()
  если винни.наелся=.л.
    если винни_возле_кролика
      инфо 'Кролик, давай поедим твой мёд.'
      если кролик.характер=0
        инфо 'Я не хочу разговаривать с тобой'
      иначе
        если кролик.мед
          инфо 'У меня есть мёд и я тебя угощу'
          винни.наелся:=.и.
          мелодия '\examples\wav\vin1.wav'
        иначе
          инфо 'У меня нет мёда'
        всё
      всё
    всё
  всё
кон проц

проц поздороваться_с_кроликом()
  если винни_возле_кролика
    кролик.характер:=кролик.характер+1
    инфо 'Привет Кролик.'
    инфо 'Здравствуй Пух.'
  всё
кон проц

проц поздороваться_с_ослом()

проц поздороваться_с_совой()

проц идти_вниз()
  буф:=сверху+высота
  для и от сверху до сверху+высота шаг скорость
    сверху:=и
  кц
  сверху:=буф
кон проц
